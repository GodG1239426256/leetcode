class Solution {
public:
    int countNodes(TreeNode* root) {
        TreeNode* l = root;
        TreeNode* r = root;
        int hl = 0, hr = 0;
        while(l){
            l = l->left;
            hl++;
        }
        while(r){
            r = r->right;
            hr++;
        }
        if(hl == hr)
            return pow(2, hl) - 1;
        return 1 + countNodes(root->left) + countNodes(root->right);
    }
};
